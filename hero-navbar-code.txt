===== components/ui/Nav.tsx =====
"use client";

import Link from "next/link";
import Image from "next/image";
import { AnimatePresence, motion } from "framer-motion";
import { useState } from "react";
import styles from "./Nav.module.css";

const NAV_ITEMS = ["Ritual", "Collection", "Our Story"];

export default function Nav(): JSX.Element {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <nav className={styles.nav} aria-label="Primary">
      <Link className={styles.logo} href="/" aria-label="Verrdelle home">
        <Image
          src="/images/Verdelle By EZMA.png"
          alt="Verrdelle"
          width={500}
          height={140}
          className={styles.logoImg}
          priority
        />
      </Link>
      <div className={styles.desktop}>
        {NAV_ITEMS.map((item) => (
          <Link key={item} href="#" className={styles.item} aria-label={item}>
            {item}
          </Link>
        ))}
        <Link href="#" className={`${styles.item} ${styles.shop}`} aria-label="Shop">
          Shop
        </Link>
      </div>
      <button
        className={styles.hamburger}
        onClick={() => setIsOpen((value) => !value)}
        aria-label="Toggle menu"
        aria-expanded={isOpen}
        aria-controls="mobile-nav"
        type="button"
      >
        {isOpen ? "Ã—" : "â˜°"}
      </button>
      <AnimatePresence>
        {isOpen && (
          <motion.div
            id="mobile-nav"
            className={styles.drawer}
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
          >
            {NAV_ITEMS.map((item) => (
              <Link
                key={item}
                href="#"
                className={styles.drawerItem}
                aria-label={item}
                onClick={() => setIsOpen(false)}
              >
                {item}
              </Link>
            ))}
            <Link
              href="#"
              className={`${styles.drawerItem} ${styles.shop}`}
              aria-label="Shop"
              onClick={() => setIsOpen(false)}
            >
              Shop
            </Link>
          </motion.div>
        )}
      </AnimatePresence>
    </nav>
  );
}
